But how does it look?

1. Load the events


    # In aggregate_repository.rb
    aggregate = @aggregate_class.new(aggregate_id)
    events = store.load_events(
      @aggregate_class,
      aggregate_id
    )
    aggregate.rebuild(events)

2. Build the aggregate


        # In editor/story.rb
        def apply_story_revision_created(event)
          @story_content = event.story_content
          @saved_at = event.event_published_at
        end

3. Do the thing


        # In editor/story.rb
        def publish(user_id:)
          send_to_followers!
          send_push!

          apply Editor::StoryPublished.new(
            user_id: user_id,
            story_id: story_id,
            story_content: story_content.to_h,
            published_at: Time.now
          )
        end
          

          # In editor/story.rb
          def apply_story_published(event)
            @status = :published
            @published_at = event.published_at
            @publish_at = nil
          end

4. Save the events


              # In command_handler.rb

              EventStore.store_events(
                aggregate_class,
                aggregate_id,
                aggregate.unpublished_events,
                initial_version,
                queue
              )
